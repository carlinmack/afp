lemma deriv_left_inverse:
  assumes "f holomorphic_on S" and "g holomorphic_on T"
      and "open S" and "open T"
      and "f ` S ‚äÜ T"
      and [simp]: "‚ãÄz. z ‚àà S ‚üπ g (f z) = z"
      and "w ‚àà S"
    shows "deriv f w * deriv g (f w) = 1"
proof -
  have "deriv f w * deriv g (f w) = deriv g (f w) * deriv f w"
    by (simp add: algebra_simps)
  also have "... = deriv (g o f) w"
    using assms
    by (metis analytic_on_imp_differentiable_at analytic_on_open deriv_chain image_subset_iff)
  also have "... = deriv id w"
    apply (rule complex_derivative_transform_within_open [where s=S])
    apply (rule assms holomorphic_on_compose_gen holomorphic_intros)+
    apply simp
    done
  also have "... = 1"
    using higher_deriv_id [of 1] by simp
  finally show ?thesis .
qed

lemma Cauchy_higher_deriv_bound:
    assumes holf: "f holomorphic_on (ball z r)"
        and contf: "continuous_on (cball z r) f"
        and "0 < r" and "0 < n"
        and fin : "‚ãÄw. w ‚àà ball z r ‚üπ f w ‚àà ball y B0"
      shows "norm ((deriv ^^ n) f z) ‚â§ (fact n) * B0 / r^n"
proof -
  have "0 < B0" using ‚Äπ0 < r‚Ä∫ fin [of z]
    by (metis ball_eq_empty ex_in_conv fin not_less)
  have le_B0: "‚ãÄw. cmod (w - z) ‚â§ r ‚üπ cmod (f w - y) ‚â§ B0"
    apply (rule continuous_on_closure_norm_le [of "ball z r" "Œªw. f w - y"])
    apply (auto simp: ‚Äπ0 < r‚Ä∫  dist_norm norm_minus_commute)
    apply (rule continuous_intros contf)+
    using fin apply (simp add: dist_commute dist_norm less_eq_real_def)
    done
  have "(deriv ^^ n) f z = (deriv ^^ n) (Œªw. f w) z - (deriv ^^ n) (Œªw. y) z"
    using ‚Äπ0 < n‚Ä∫ by simp
  also have "... = (deriv ^^ n) (Œªw. f w - y) z"
    by (rule higher_deriv_diff [OF holf, symmetric]) (auto simp: ‚Äπ0 < r‚Ä∫ holomorphic_on_const)
  finally have "(deriv ^^ n) f z = (deriv ^^ n) (Œªw. f w - y) z" .
  have contf': "continuous_on (cball z r) (Œªu. f u - y)"
    by (rule contf continuous_intros)+
  have holf': "(Œªu. (f u - y)) holomorphic_on (ball z r)"
    by (simp add: holf holomorphic_on_diff holomorphic_on_const)
  define a where "a = (2 * pi)/(fact n)"
  have "0 < a"  by (simp add: a_def)
  have "B0/r^(Suc n)*2 * pi * r = a*((fact n)*B0/r^n)"
    using ‚Äπ0 < r‚Ä∫ by (simp add: a_def divide_simps)
  have der_dif: "(deriv ^^ n) (Œªw. f w - y) z = (deriv ^^ n) f z"
    using ‚Äπ0 < r‚Ä∫ ‚Äπ0 < n‚Ä∫
    by (auto simp: higher_deriv_diff [OF holf holomorphic_on_const])
  have "norm ((2 * of_real pi * ùóÇ)/(fact n) * (deriv ^^ n) (Œªw. f w - y) z)
        ‚â§ (B0/r^(Suc n)) * (2 * pi * r)"
    apply (rule has_contour_integral_bound_circlepath [of "(Œªu. (f u - y)/(u - z)^(Suc n))" _ z])
    using Cauchy_has_contour_integral_higher_derivative_circlepath [OF contf' holf']
    using ‚Äπ0 < B0‚Ä∫ ‚Äπ0 < r‚Ä∫
    apply (auto simp: norm_divide norm_mult norm_power divide_simps le_B0)
    done
  then show ?thesis
    using ‚Äπ0 < r‚Ä∫
    by (auto simp: norm_divide norm_mult norm_power field_simps der_dif le_B0)
qed

lemma higher_deriv_comp_lemma:
    assumes s: "open s" and holf: "f holomorphic_on s"
        and "z ‚àà s"
        and t: "open t" and holg: "g holomorphic_on t"
        and fst: "f ` s ‚äÜ t"
        and n: "i ‚â§ n"
        and dfz: "deriv f z = 1" and zero: "‚ãÄi. ‚ü¶1 < i; i ‚â§ n‚üß ‚üπ (deriv ^^ i) f z = 0"
      shows "(deriv ^^ i) (g o f) z = (deriv ^^ i) g (f z)"
using n holg
proof (induction i arbitrary: g)
  case 0 then show ?case by simp
next
  case (Suc i)
  have "g ‚àò f holomorphic_on s" using "Suc.prems" holf
    using fst  by (simp add: holomorphic_on_compose_gen image_subset_iff)
  then have 1: "deriv (g ‚àò f) holomorphic_on s"
    by (simp add: holomorphic_deriv s)
  have dg: "deriv g holomorphic_on t"
    using Suc.prems by (simp add: Suc.prems(2) holomorphic_deriv t)
  then have "deriv g holomorphic_on f ` s"
    using fst  by (simp add: holomorphic_on_subset image_subset_iff)
  then have dgf: "(deriv g o f) holomorphic_on s"
    by (simp add: holf holomorphic_on_compose)
  then have 2: "(Œªw. (deriv g o f) w * deriv f w) holomorphic_on s"
    by (blast intro: holomorphic_intros holomorphic_on_compose holf s)
  have "(deriv ^^ i) (deriv (g o f)) z = (deriv ^^ i) (Œªw. deriv g (f w) * deriv f w) z"
    apply (rule higher_deriv_transform_within_open [OF 1 2 [unfolded o_def] s ‚Äπz ‚àà s‚Ä∫])
    apply (rule deriv_chain)
    using holf Suc.prems fst apply (auto simp: holomorphic_on_imp_differentiable_at s t)
    done
  also have "... = (‚àëj=0..i. of_nat(i choose j) * (deriv ^^ j) (Œªw. deriv g (f w)) z * (deriv ^^ (i - j)) (deriv f) z)"
    apply (rule higher_deriv_mult [OF dgf [unfolded o_def] _ s ‚Äπz ‚àà s‚Ä∫])
    by (simp add: holf holomorphic_deriv s)
  also have "... = (‚àëj=i..i. of_nat(i choose j) * (deriv ^^ j) (Œªw. deriv g (f w)) z * (deriv ^^ Suc (i - j)) f z)"
  proof -
    have *: "(deriv ^^ j) (Œªw. deriv g (f w)) z = 0"  if "j < i" and nz: "(deriv ^^ (i - j)) (deriv f) z ‚â† 0" for j
    proof -
      have "1 < Suc (i - j)" "Suc (i - j) ‚â§ n"
        using ‚Äπj < i‚Ä∫ ‚ÄπSuc i ‚â§ n‚Ä∫ by auto
      then show ?thesis  by (metis comp_def funpow.simps(2) funpow_swap1 zero nz)
    qed
    then show ?thesis
      apply (simp only: funpow_Suc_right o_def)
      apply (rule comm_monoid_add_class.sum.mono_neutral_right, auto)
      done
  qed
  also have "... = (deriv ^^ i) (deriv g) (f z)"
    using Suc.IH [OF _ dg] Suc.prems by (simp add: dfz)
  finally show ?case
    by (simp only: funpow_Suc_right o_def)
qed


lemma higher_deriv_comp_iter_lemma:
    assumes s: "open s" and holf: "f holomorphic_on s"
        and fss: "f ` s ‚äÜ s"
        and "z ‚àà s" and [simp]: "f z = z"
        and n: "i ‚â§ n"
        and dfz: "deriv f z = 1" and zero: "‚ãÄi. ‚ü¶1 < i; i ‚â§ n‚üß ‚üπ (deriv ^^ i) f z = 0"
      shows "(deriv ^^ i) (f^^m) z = (deriv ^^ i) f z"
proof -
  have holfm: "(f^^m) holomorphic_on s" for m
    apply (induction m, simp add: holomorphic_on_ident)
    apply (simp only: funpow_Suc_right holomorphic_on_compose_gen [OF holf _ fss])
    done
  show ?thesis using n
  proof (induction m)
    case 0 with dfz show ?case
      by (auto simp: zero)
  next
    case (Suc m)
    have "(deriv ^^ i) (f ^^ m ‚àò f) z = (deriv ^^ i) (f ^^ m) (f z)"
      using Suc.prems holfm ‚Äπz ‚àà s‚Ä∫ dfz fss higher_deriv_comp_lemma holf s zero by blast
    also have "... = (deriv ^^ i) f z"
      by (simp add: Suc)
    finally show ?case
      by (simp only: funpow_Suc_right)
  qed
qed

lemma higher_deriv_iter_top_lemma:
    assumes s: "open s" and holf: "f holomorphic_on s"
        and fss: "f ` s ‚äÜ s"
        and "z ‚àà s" and [simp]: "f z = z"
        and dfz [simp]: "deriv f z = 1"
        and n: "1 < n" "‚ãÄi. ‚ü¶1 < i; i < n‚üß ‚üπ (deriv ^^ i) f z = 0"
      shows "(deriv ^^ n) (f ^^ m) z = m * (deriv ^^ n) f z"
using n
proof (induction n arbitrary: m)
  case 0 then show ?case by simp
next
  case (Suc n)
  have [simp]: "(f^^m) z = z" for m
    by (induction m) auto
  have fms_sb: "(f^^m) ` s ‚äÜ s" for m
    apply (induction m)
    using fss
    apply force+
    done
  have holfm: "(f^^m) holomorphic_on s" for m
    apply (induction m, simp add: holomorphic_on_ident)
    apply (simp only: funpow_Suc_right holomorphic_on_compose_gen [OF holf _ fss])
    done
  then have holdfm: "deriv (f ^^ m) holomorphic_on s" for m
    by (simp add: holomorphic_deriv s)
  have holdffm: "(Œªz. deriv f ((f ^^ m) z)) holomorphic_on s" for m
    apply (rule holomorphic_on_compose_gen [where g="deriv f" and t=s, unfolded o_def])
    using s ‚Äπz ‚àà s‚Ä∫ holfm holf fms_sb by (auto intro: holomorphic_intros)
  have f_cd_w: "‚ãÄw. w ‚àà s ‚üπ f field_differentiable at w"
    using holf holomorphic_on_imp_differentiable_at s by blast
  have f_cd_mw: "‚ãÄm w. w ‚àà s ‚üπ (f^^m) field_differentiable at w"
    using holfm holomorphic_on_imp_differentiable_at s by auto
  have der_fm [simp]: "deriv (f ^^ m) z = 1" for m
    apply (induction m, simp add: deriv_ident)
    apply (subst funpow_Suc_right)
    apply (subst deriv_chain)
    using ‚Äπz ‚àà s‚Ä∫ holfm holomorphic_on_imp_differentiable_at s f_cd_w apply auto
    done
  note Suc(3) [simp]
  note n_Suc = Suc
  show ?case
  proof (induction m)
    case 0 with n_Suc show ?case
      by (metis Zero_not_Suc funpow_simps_right(1) higher_deriv_id lambda_zero nat_neq_iff of_nat_0)
  next
    case (Suc m)
    have deriv_nffm: "(deriv ^^ n) (deriv f o (f ^^ m)) z = (deriv ^^ n) (deriv f) ((f ^^ m) z)"
      apply (rule higher_deriv_comp_lemma [OF s holfm ‚Äπz ‚àà s‚Ä∫ s _ fms_sb order_refl])
      using ‚Äπz ‚àà s‚Ä∫ fss higher_deriv_comp_iter_lemma holf holf holomorphic_deriv s
        apply auto
      done
    have "deriv (f ^^ m ‚àò f) holomorphic_on s"
      by (metis funpow_Suc_right holdfm)
    moreover have "(Œªw. deriv f ((f ^^ m) w) * deriv (f ^^ m) w) holomorphic_on s"
      by (rule holomorphic_on_mult [OF holdffm holdfm])
    ultimately have "(deriv ^^ n) (deriv (f ^^ m ‚àò f)) z = (deriv ^^ n) (Œªw. deriv f ((f ^^ m) w) * deriv (f ^^ m) w) z"
      apply (rule higher_deriv_transform_within_open [OF _ _ s ‚Äπz ‚àà s‚Ä∫])
      by (metis comp_funpow deriv_chain f_cd_mw f_cd_w fms_sb funpow_swap1 image_subset_iff o_id)
    also have "... =
          (‚àëi=0..n. of_nat(n choose i) * (deriv ^^ i) (Œªw. deriv f ((f ^^ m) w)) z *
                     (deriv ^^ (n - i)) (deriv (f ^^ m)) z)"
      by (rule higher_deriv_mult [OF holdffm holdfm s ‚Äπz ‚àà s‚Ä∫])
    also have "... = (‚àëi ‚àà {0,n}. of_nat(n choose i) * (deriv ^^ i) (Œªw. deriv f ((f ^^ m) w)) z *
                     (deriv ^^ (n - i)) (deriv (f ^^ m)) z)"
    proof -
      have *: "(deriv ^^ i) (Œªw. deriv f ((f ^^ m) w)) z = 0"  if "i ‚â§ n" "0 < i" "i ‚â† n" and nz: "(deriv ^^ (n - i)) (deriv (f ^^ m)) z ‚â† 0" for i
      proof -
        have less: "1 < Suc (n-i)" and le: "Suc (n-i) ‚â§ n"
          using that by auto
        have "(deriv ^^ (Suc (n - i))) (f ^^ m) z = (deriv ^^(Suc (n - i))) f z"
          apply (rule higher_deriv_comp_iter_lemma [OF s holf fss ‚Äπz ‚àà s‚Ä∫ ‚Äπf z = z‚Ä∫ le dfz])
          by simp
        also have "... = 0"
          using n_Suc(3) less le le_imp_less_Suc by blast
        finally have "(deriv ^^ (Suc (n - i))) (f ^^ m) z = 0" .
        then show ?thesis by (simp add: funpow_swap1 nz)
      qed
      show ?thesis
        by (rule comm_monoid_add_class.sum.mono_neutral_right) (auto simp: *)
    qed
    also have "... = of_nat (Suc m) * (deriv ^^ n) (deriv f) z"
      apply (subst Groups_Big.comm_monoid_add_class.sum.insert)
      apply (simp_all add: deriv_nffm [unfolded o_def] of_nat_Suc [of 0] del: of_nat_Suc)
      using n_Suc(2) Suc
      apply (auto simp del: funpow.simps simp: algebra_simps funpow_simps_right)
      done
    finally have "(deriv ^^ n) (deriv (f ^^ m ‚àò f)) z = of_nat (Suc m) * (deriv ^^ n) (deriv f) z" .
    then show ?case
      apply (simp only: funpow_Suc_right)
      apply (simp add: o_def del: of_nat_Suc)
      done
  qed
qed

theorem first_Cartan_dim_1:
    assumes holf: "f holomorphic_on s"
        and "open s" "connected s" "bounded s"
        and fss: "f ` s ‚äÜ s"
        and "z ‚àà s" and [simp]: "f z = z"
        and dfz [simp]: "deriv f z = 1"
        and "w ‚àà s"
      shows "f w = w"
proof -
  obtain c where "0 < c" and c: "s ‚äÜ ball z c"
    using ‚Äπbounded s‚Ä∫ bounded_subset_ballD by blast
  obtain r where "0 < r" and r: "cball z r ‚äÜ s"
    using ‚Äπz ‚àà s‚Ä∫ open_contains_cball ‚Äπopen s‚Ä∫ by blast
  then have bzr: "ball z r ‚äÜ s" using ball_subset_cball by blast
  have fms_sb: "(f^^m) ` s ‚äÜ s" for m
    apply (induction m)
    using fss apply force+
    done
  have holfm: "(f^^m) holomorphic_on s" for m
    apply (induction m, simp add: holomorphic_on_ident)
    apply (simp only: funpow_Suc_right holomorphic_on_compose_gen [OF holf _ fss])
    done
  have *: "(deriv ^^ n) f z = (deriv ^^ n) id z" for n
  proof -
    consider "n = 0" | "n = 1" | "1 < n" by arith
    then show ?thesis
    proof cases
      assume "n = 0" then show ?thesis by force
    next
      assume "n = 1" then show ?thesis by force
    next
      assume n1: "n > 1"
      then have "(deriv ^^ n) f z = 0"
      proof (induction n rule: less_induct)
        case (less n)
        have le: "real m * cmod ((deriv ^^ n) f z) ‚â§ fact n * c / r ^ n" if "m‚â†0" for m
        proof -
          have holfm': "(f ^^ m) holomorphic_on ball z r"
            using holfm bzr holomorphic_on_subset by blast
          then have contfm': "continuous_on (cball z r) (f ^^ m)"
            using ‚Äπcball z r ‚äÜ s‚Ä∫ holfm holomorphic_on_imp_continuous_on holomorphic_on_subset by blast
          have "real m * cmod ((deriv ^^ n) f z) = cmod (real m * (deriv ^^ n) f z)"
            by (simp add: norm_mult)
          also have "... = cmod ((deriv ^^ n) (f ^^ m) z)"
            apply (subst higher_deriv_iter_top_lemma [OF ‚Äπopen s‚Ä∫ holf fss ‚Äπz ‚àà s‚Ä∫ ‚Äπf z = z‚Ä∫ dfz])
            using less apply auto
            done
          also have "... ‚â§ fact n * c / r ^ n"
            apply (rule Cauchy_higher_deriv_bound [OF holfm' contfm' ‚Äπ0 < r‚Ä∫, where y=z])
            using less.prems apply linarith
            using fms_sb c r ball_subset_cball
            apply blast
            done
          finally show ?thesis .
        qed
        have "cmod ((deriv ^^ n) f z) = 0"
          apply (rule real_archimedian_rdiv_eq_0 [where c = "(fact n) * c / r ^ n"])
          apply simp
          using ‚Äπ0 < r‚Ä∫ ‚Äπ0 < c‚Ä∫
          apply (simp add: divide_simps)
          apply (blast intro: le)
          done
        then show ?case by simp
      qed
      with n1 show ?thesis by simp
    qed
  qed
  have "f w = id w"
    by (rule holomorphic_fun_eq_on_connected
                 [OF holf holomorphic_on_id ‚Äπopen s‚Ä∫ ‚Äπconnected s‚Ä∫ * ‚Äπz ‚àà s‚Ä∫ ‚Äπw ‚àà s‚Ä∫])
  also have "... = w" by simp
  finally show ?thesis .
qed

lemma Cartan_is_linear:
  assumes holf: "f holomorphic_on s"
      and "open s" and "connected s"
      and "0 ‚àà s"
      and ins: "‚ãÄu z. ‚ü¶norm u = 1; z ‚àà s‚üß ‚üπ u * z ‚àà s"
      and feq: "‚ãÄu z. ‚ü¶norm u = 1; z ‚àà s‚üß ‚üπ f (u * z) = u * f z"
    shows "‚àÉc. ‚àÄz ‚àà s. f z = c * z"
proof -
  have [simp]: "f 0 = 0"
    using feq [of "-1" 0] assms by simp
  have uneq: "u^n * (deriv ^^ n) f (u * z) = u * (deriv ^^ n) f z"
       if "norm u = 1" "z ‚àà s" for n u z
  proof -
    have holfuw: "(Œªw. f (u * w)) holomorphic_on s"
      apply (rule holomorphic_on_compose_gen [OF _ holf, unfolded o_def])
      using that ins by (auto simp: holomorphic_on_linear)
    have hol_d_fuw: "(deriv ^^ n) (Œªw. u * f w) holomorphic_on s" for n
      by (rule holomorphic_higher_deriv holomorphic_intros holf assms)+
    have *: "(deriv ^^ n) (Œªw. u * f w) z = u * (deriv ^^ n) f z" if "z ‚àà s" for z
    using that
    proof (induction n arbitrary: z)
      case 0 then show ?case by simp
    next
      case (Suc n)
      have "deriv ((deriv ^^ n) (Œªw. u * f w)) z = deriv (Œªw. u * (deriv ^^ n) f w) z"
        apply (rule complex_derivative_transform_within_open [OF hol_d_fuw])
        apply (auto intro!: holomorphic_higher_deriv holomorphic_intros assms Suc)
        done
      also have "... = u * deriv ((deriv ^^ n) f) z"
        apply (rule deriv_cmult)
        using Suc ‚Äπopen s‚Ä∫ holf holomorphic_higher_deriv holomorphic_on_imp_differentiable_at by blast
      finally show ?case by simp
    qed
    have "(deriv ^^ n) (Œªw. f (u * w)) z = u ^ n * (deriv ^^ n) f (u * z)"
      apply (rule higher_deriv_compose_linear [OF holf ‚Äπopen s‚Ä∫ ‚Äπopen s‚Ä∫])
      apply (simp add: that)
      apply (simp add: ins that)
      done
    moreover have "(deriv ^^ n) (Œªw. f (u * w)) z = u * (deriv ^^ n) f z"
      apply (subst higher_deriv_transform_within_open [OF holfuw, of "Œªw. u * f w"])
      apply (rule holomorphic_intros holf assms that)+
      apply blast
      using * ‚Äπz ‚àà s‚Ä∫ apply blast
      done
    ultimately show ?thesis by metis
  qed
  have dnf0: "(deriv ^^ n) f 0 = 0" if len: "2 ‚â§ n" for n
  proof -
    have **: "z = 0" if "‚ãÄu::complex. norm u = 1 ‚üπ u ^ n * z = u * z" for z
    proof -
      have "‚àÉu::complex. norm u = 1 ‚àß u ^ n ‚â† u"
        using complex_not_root_unity [of "n-1"] len
        apply (simp add: algebra_simps le_diff_conv2, clarify)
        apply (rule_tac x=u in exI)
        apply (subst (asm) power_diff)
        apply auto
        done
      with that show ?thesis
        by auto
    qed
    show ?thesis
      apply (rule **)
      using uneq [OF _ ‚Äπ0 ‚àà s‚Ä∫]
      by force
  qed
  show ?thesis
    apply (rule_tac x = "deriv f 0" in exI, clarify)
    apply (rule holomorphic_fun_eq_on_connected [OF holf _ ‚Äπopen s‚Ä∫ ‚Äπconnected s‚Ä∫ _ ‚Äπ0 ‚àà s‚Ä∫])
    using dnf0 apply (auto simp: holomorphic_on_linear)
    done
qed

theorem second_Cartan_dim_1:
  assumes holf: "f holomorphic_on ball 0 r"
      and holg: "g holomorphic_on ball 0 r"
      and [simp]: "f 0 = 0" and [simp]: "g 0 = 0"
      and ballf: "‚ãÄz. z ‚àà ball 0 r ‚üπ f z ‚àà ball 0 r"
      and ballg: "‚ãÄz. z ‚àà ball 0 r ‚üπ g z ‚àà ball 0 r"
      and fg: "‚ãÄz. z ‚àà ball 0 r ‚üπ f (g z) = z"
      and gf: "‚ãÄz. z ‚àà ball 0 r ‚üπ g (f z) = z"
      and "0 < r"
    shows "‚àÉt. ‚àÄz ‚àà ball 0 r. g z = exp(ùóÇ * of_real t) * z"
proof -
  have c_le_1: "c ‚â§ 1"
    if "0 ‚â§ c" "‚ãÄx. 0 ‚â§ x ‚üπ x < r ‚üπ c * x < r" for c
  proof -
    have rst: "‚ãÄr s t::real. 0 = r ‚à® s/r < t ‚à® r < 0 ‚à® ¬¨ s < r * t"
      by (metis (no_types) mult_less_cancel_left_disj nonzero_mult_div_cancel_left times_divide_eq_right)
    { assume "¬¨ r < c ‚àß c * (c * (c * (c * r))) < 1"
     then have "1 ‚â§ c ‚üπ (‚àÉr. ¬¨ 1 < r ‚àß ¬¨ r < c)"

  have ugeq: "u * g z = g (u * z)" if nou: "norm u = 1" and z: "z ‚àà ball 0 r" for u z

      apply (subst deriv_chain [where g=f and f = "Œªa. (u * g a)", unfolded o_def])

      done

      apply (subst (asm) gf)
      apply (simp add: dist_norm norm_mult nou)
      apply (metis ballg mem_ball_0 mult.left_neutral norm_mult nou z, simp)

  qed
  obtain c where c: "‚ãÄz. z ‚àà ball 0 r ‚üπ g z = c * z"

  have 11: "1 / norm c ‚â§ 1"
