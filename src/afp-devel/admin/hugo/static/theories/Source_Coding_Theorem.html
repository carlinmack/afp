  assumes X: "simple_distributed M X Px"

    simple_distributed_nonneg[OF X]

  by (simp add:  ac_simps)

abbreviation cw_len_concat :: "'b word ‚áí nat" where
  "cw_len_concat w ‚â° foldr (Œªx s. (cw_len x) + s) w 0"

lemma cw_len_length: "cw_len_concat w = length (enc w)"
proof (induction w)
    case Nil
    show ?case using real_code by simp
    case (Cons a w)
    have "cw_len_concat (a # w) = cw_len a + cw_len_concat w" by simp

      by (metis length_append list.distinct(1) list.sel(1) list.sel(3))
qed

lemma k_words_rel:
  "k_words (Suc k) = {w. (hd w ‚àà L ‚àß tl w ‚àà k_words k ‚àß w ‚â† [])}"

    fix k
    show "k_words (Suc k) ‚äÜ {w. (hd w ‚àà L ‚àß tl w ‚àà k_words k ‚àß w ‚â† [] )}" (is "?l ‚äÜ ?r")
  proof
      fix w
      assume w_kw: "w ‚àà k_words (Suc k)"
      hence "real_word w" by simp
      hence "hd w ‚àà L"
        by (metis (mono_tags) w_kw hd_in_set list.size(3) mem_Collect_eq nat.distinct(1) subset_code(1))
      moreover have "length w = Suc k" using w_kw by simp
      moreover hence "w ‚â† []" by auto
      moreover have "real_word (tl w)" using ‚Äπreal_word w‚Ä∫ calculation(3) rw_tail by auto
      ultimately show "w ‚àà ?r" using w_kw by simp
  qed
next
    fix k
    show "k_words (Suc k) ‚äá {w. (hd w ‚àà L ‚àß tl w ‚àà k_words k ‚àß w ‚â† [])}"
  proof
      fix w
      assume asm: "w ‚àà {w. hd w ‚àà L ‚àß tl w ‚àà {w. length w = k ‚àß real_word w} ‚àß w ‚â† []}"
      hence "hd w ‚àà L ‚àß length (tl w) = k ‚àß real_word (tl w)" by simp
      hence "real_word w"
        by (metis empty_iff insert_subset list.collapse list.set(1) set_simps(2) subsetI)
      moreover hence "length w = Suc k" using asm by auto
      ultimately show "w ‚àà k_words (Suc k)" by simp
  qed
qed

lemma bij_k_words:

    unfolding bij_betw_def

    let ?f = "(Œªwi. Cons (fst wi) (snd wi))"
    let ?S = "L √ó (k_words k)"
    let ?T = "k_words (Suc k)"
    show "inj_on ?f ?S" by (simp add: inj_on_def)
    show "?f`?S = ?T"
  proof (rule ccontr)
      assume "?f ` ?S ‚â† ?T"
      hence "‚àÉw. w‚àà ?T ‚àß w ‚àâ ?f`?S" by auto
      then obtain w where asm: "w‚àà ?T ‚àß w ‚àâ ?f`?S" by blast
      hence "w = ?f (hd w,tl w)" using k_words_rel by simp
      moreover have "(hd w,tl w) ‚àà ?S" using k_words_rel asm by simp
      ultimately have "w ‚àà ?f`?S" by blast
      thus "False" using asm by simp
  qed
qed

lemma cartesian_product:
  fixes f::"('c ‚áí real)"
  fixes g::"('d ‚áí real)"
  assumes "finite A"
  assumes "finite B"
shows "(‚àëb‚ààB. g b) * (‚àëa‚ààA. f a) = (‚àëab‚ààA√óB. f (fst ab) * g (snd ab))"
    using bilinear_times bilinear_sum[where h="(Œªx y. x * y)" and f="f" and g="g"] assms
    by (metis (erased, lifting) sum.cong split_beta' Groups.ab_semigroup_mult_class.mult.commute)

    case 0
    have "k_words 0 = {[]}" by auto
    thus ?case by simp
next
    case (Suc n)
    have " ùí¶ ^Suc n = ùí¶ ^n * ùí¶ " by simp
    also have "‚Ä¶ = (‚àëw ‚àà k_words n. 1 / b^cw_len_concat w) * (‚àëi‚ààL. 1 / b^cw_len i)"
      using Suc.hyps ùí¶_def by auto
    also have "‚Ä¶ = (‚àëwi ‚àà L √ó k_words n. 1/b^cw_len (fst wi) * (1 / b^cw_len_concat (snd wi)))"
      using fin_L finite_k_words cartesian_product
      by blast
    also have "‚Ä¶ = (‚àëwi ‚àà L √ó k_words n. 1 / b^(cw_len_concat (snd wi) + cw_len (fst wi)))"
      by (metis (no_types, lifting) power_add add.commute power_one_over)
    also have "‚Ä¶ = (‚àëwi ‚àà L √ó k_words n. 1 / b^cw_len_concat (fst wi # snd wi))"

    also have "‚Ä¶ = (‚àëw ‚àà (k_words (Suc n)). 1 / b^(cw_len_concat w))"
      using bij_k_words sum.reindex_bij_betw by fastforce
    finally show ?case by simp
qed

lemma sum_vimage_proof:
  fixes g::"nat ‚áí real"
  assumes "‚ãÄw. f w < bd"
shows "finite S ‚üπ (‚àëw‚ààS. g (f w)) = (‚àë m=0..<bd. (card ((f-`{m}) ‚à© S) )* g m)"
(is "_ ‚üπ _ = (‚àë m=0..<bd. ?ff m S)")
proof (induct S rule: finite_induct)
    case empty
    show ?case by simp
next
    case (insert x F)
    let ?rr = "(‚àëm = 0..<bd. ?ff m (insert x F))"
    have "(f x) ‚àà {0..<bd}" using assms by simp
    hence "‚ãÄh::(nat ‚áí real). (‚àëm=0..<bd. h m) = (‚àëy‚àà({0..<bd} - {f x}).h y) + h (f x)"
      by (metis diff_add_cancel finite_atLeastLessThan sum_diff1)
    moreover hence
    "(‚àëm = 0..<bd. ?ff m (insert x F))
    = (‚àëm‚àà{0..<bd} - {f x}. ?ff m (insert x F)) + card (f -` {f x} ‚à© F) * g (f x) + g (f x)"
      by (simp add: semiring_normalization_rules(2), simp add: insert)
    ultimately have "(‚àëm = 0..<bd. ?ff m (insert x F)) = (‚àëm‚àà{0..<bd}. ?ff m F) + g (f x)"
      by fastforce
    thus ?case using insert by simp
qed

lemma sum_vimage:
  fixes g::"nat ‚áí real"
  assumes bounded: "‚ãÄw. w ‚àà S ‚üπ f w < bd" and "0 < bd"
  assumes finite: "finite S"
shows "(‚àëw‚ààS. g (f w)) = (‚àë m=0..<bd. (card ((f-`{m}) ‚à© S) ) * g m)"
(is "?s1 = ?s2")
proof -
    let ?ff = "(Œªx. if x‚ààS then f x else 0)"
    let ?ss1 = "(‚àëw‚ààS. g (?ff w))"
    let ?ss2 = "(‚àë m=0..<bd. (card ((?ff-`{m}) ‚à© S) ) * g m)"
    have "?s1 =?ss1" by simp
    moreover have"‚ãÄm. ?ff -`{m} ‚à© S = f-`{m} ‚à© S" by auto
    moreover hence "?s2 = ?ss2" by simp
    moreover have "‚ãÄw . ?ff w < bd" using assms by simp
    moreover hence "?ss1 = ?ss2" using sum_vimage_proof[of "?ff"] finite by blast
    ultimately show "?s1 = ?s2" by metis
qed

lemma empty_set_k_words:

shows "set_of_k_words_length_m k 0 = {}"

definition KL_div ::"'b set ‚áí ('b ‚áí real) ‚áí ('b ‚áí real) ‚áí real" where
  "KL_div S a d = (‚àë i ‚àà S. a i * log b (a i / d i))"

lemma KL_div_mul:
  assumes "0 < d" "d ‚â§ 1"
  assumes "‚ãÄi. i‚ààS ‚üπ 0 ‚â§ a i"
  assumes "‚ãÄi. i‚ààS ‚üπ 0 < e i"
shows "KL_div S a e ‚â• KL_div S a (Œªi. e i / d)"
    unfolding KL_div_def
proof -
    {
    fix i
    assume "i‚ààS"
    hence "a i / (e i / d) ‚â§ a i / e i" using assms
      by (metis (no_types) div_by_1 frac_le less_imp_triv not_less)
    hence "log b (a i / (e i / d)) ‚â§ log b (a i / e i)" using assms(1)
      by (metis (full_types) b_gt_1 divide_divide_eq_left inverse_divide le_less_linear log_le
    log_neg_const order_refl times_divide_eq_right zero_less_mult_iff)
    }
    thus "(‚àëi‚ààS. a i * log b (a i / (e i / d))) ‚â§ (‚àëi‚ààS. a i * log b (a i / e i))"
      by (meson mult_left_mono assms sum_mono)
qed

lemma KL_div_pos:
  fixes a e::"'b ‚áí real"
  assumes fin: "finite S"
  assumes nemp: "S ‚â† {}"
  assumes non_null: "‚ãÄi. i‚ààS ‚üπ 0 < a i" "‚ãÄi. i‚àà S ‚üπ 0 < e i"
  assumes sum_a_one: "(‚àë i ‚àà S. a i) = 1"
  assumes sum_c_one: "(‚àë i ‚àà S. e i) = 1"
shows "0 ‚â§ KL_div S a e"
    unfolding KL_div_def
proof -
    let ?f = "Œªi. e i / a i"
    have f_pos: "‚ãÄi. i‚ààS ‚üπ 0 < ?f i"
      using non_null
      by simp
    have a_pos: "‚ãÄi. i‚àà S ‚üπ 0 ‚â§ a i"
      using non_null
      by (simp add: order.strict_implies_order)
    have "- log b (‚àëi‚ààS. a i * e i / a i) ‚â§ (‚àëi‚ààS. a i * - log b (e i / a i))"
      using convex_on_sum[OF fin nemp  minus_log_convex[OF b_gt_1] convex_real_interval(3)
                             sum_a_one a_pos, of "Œªi. e i / a i"] f_pos by simp
    also have "-log b (‚àëi‚ààS. a i * e i / a i) = -log b (‚àëi‚ààS. e i)"
  proof -
      from non_null(1) have "‚ãÄi. i ‚àà S ‚üπ a i * e i / a i = e i" by force
      thus ?thesis by simp
  qed
    finally have "0 ‚â§ (‚àëi‚ààS. a i * - log b (e i / a i))"
      by (simp add: sum_c_one)

theorem KL_div_pos2:
  fixes a d::"'b ‚áí real"
  assumes fin: "finite S"
  assumes non_null: "‚ãÄi. i‚ààS ‚üπ 0 ‚â§ a i" "‚ãÄi. i‚àà S ‚üπ 0 < d i"
  assumes sum_a_one: "(‚àë i ‚àà S. a i) = 1"
  assumes sum_c_one: "(‚àë i ‚àà S. d i) = 1"
shows "0 ‚â§ KL_div S a d"
proof -
    have "S = (S ‚à© {i. 0 < a i}) ‚à™ (S ‚à© {i. 0 = a i})" using non_null(1) by fastforce
    moreover have "(S ‚à© {i. 0 < a i}) ‚à© (S ‚à© {i. 0 = a i}) = {}" by auto
    ultimately have
    eq: "KL_div S a d = KL_div (S ‚à© {i. 0 < a i}) a d + KL_div (S ‚à© {i. 0 = a i}) a d"
      unfolding KL_div_def
      by (metis (mono_tags, lifting) fin finite_Un sum.union_disjoint)
    have "KL_div (S ‚à© {i. 0 = a i}) a d = 0" unfolding KL_div_def by simp
    hence "KL_div S a d = KL_div (S ‚à© {i. 0 < a i}) a d" using eq by simp
    moreover have "0 ‚â§ KL_div (S ‚à© {i. 0 < a i}) a d"
  proof(cases "(S ‚à© {i. 0 < a i}) = {}")
      case True
      thus ?thesis unfolding KL_div_def by simp
  next
      case False
      let ?c = "Œªi. d i / (‚àëj ‚àà(S ‚à© {i. 0 < a i}). d j)"
      have 1: "(‚ãÄi. i ‚àà S ‚à© {i. 0 < a i} ‚üπ 0 < a i)" by simp
      have 2: "(‚ãÄi. i ‚àà S ‚à© {i. 0 < a i} ‚üπ 0 < ?c i)"
        by (metis False IntD1 divide_pos_pos fin finite_Int non_null(2) sum_pos)
      have 3: "(‚àëi‚àà (S ‚à© {i. 0 < a i}). a i) = 1"
        using sum.cong[of S, of S, of "(Œªx. if x ‚àà {i. 0 < a i} then a x else 0)", of a]
      sum.inter_restrict[OF fin, of a] non_null(1) sum_a_one
        by fastforce
      have "(‚àëi‚ààS ‚à© {j. 0 < a j}. ?c i) = (‚àëi‚ààS ‚à© {j. 0 < a j}. d i) / (‚àëi‚ààS ‚à© {j. 0 < a j}. d i)"
        by (metis sum_divide_distrib)
      hence 5: "(‚àëi‚ààS ‚à© {j. 0 < a j}. ?c i) = 1" using 2 False by force
      hence "0 ‚â§ KL_div (S ‚à© {j. 0 < a j}) a ?c"
        using KL_div_pos_gen[
      OF finite_Int[OF disjI1, of S, of "{j. 0 < a j}"], of a, of ?c
      ] 1 2 3
        by (metis fin)
      have fstdb: "0 < (‚àëi‚ààS ‚à© {i. 0 < a i}. d i)" using non_null(2) False
        by (metis Int_Collect fin finite_Int sum_pos)
      have 6: "0 ‚â§ KL_div (S ‚à© {i. 0 < a i}) a (Œªi. d i / (‚àëi‚àà(S ‚à© {i. 0 < a i}). d i))"
        using 2 3 5
        KL_div_pos_gen[
      OF finite_Int[OF disjI1, OF fin], of "{i. 0 < a i}", of "a", of "?c"
      ]
        by simp
      hence
      "KL_div (S ‚à© {j. 0 < a j}) a (Œªi. d i / (‚àëi‚àà(S ‚à© {i. 0 < a i}). d i)) ‚â§ KL_div (S ‚à© {j. 0 < a j}) a d"
        using non_null sum.inter_restrict[OF fin, of d, of "{i. 0 < a i}"]
        sum_mono[of S, of "(Œªx. if x ‚àà {i. 0 < a i} then d x else 0)", of d] non_null(2) sum_c_one
        non_null(2) fstdb KL_div_mul
        by force
      moreover have "0 ‚â§ KL_div (S ‚à© {j. 0 < a j}) a (Œªi. d i / (‚àëi‚àà(S ‚à© {i. 0 < a i}). d i))"
        using KL_div_pos_gen[ OF finite_Int[OF disjI1, OF fin]] using 2 3 5 by fastforce
      ultimately show "0 ‚â§ KL_div (S ‚à© {j. 0 < a j}) a d" by simp
  qed
    ultimately show ?thesis by simp
qed
